<h5>html code for animating these assets</h5>

<pre>
  <code>

.overlay &#123;
	position: relative;
	width: 75vw; 	 
	overflow: hidden;
&#125;
.layer &#123;
	position: absolute;
	top: 0;
	left: 0;
	display: flex; /&lowast; Arrange images side-by-side &lowast;/
	width: 200%; /&lowast; Combined width of two images &lowast;/
	object-fit: cover;
&#125;

.layer img &#123;
	display: block;
	width: 50%; /&lowast; Each image takes up half the layer's width &lowast;/
	height: auto; 
&#125;

&#64;keyframes seamless-loop &#123;
	0% &#123;
		transform: translateX(0); /&lowast; Start with the first image visible &lowast;/
	&#125;
	100% &#123;
		transform: translateX(-50%); /&lowast; Move left by half of the layer's width &lowast;/
	&#125;
&#125;

/&lowast; Apply animations to each layer with different speeds &lowast;/
.layer1 &#123;
	animation: seamless-loop 105s linear infinite;
&#125;

.layer2 &#123;
	animation: seamless-loop 70s linear infinite;
&#125;

.layer3 &#123;
	animation: seamless-loop 30s linear infinite;
&#125;

.layer4 &#123;
	animation: seamless-loop 3s linear infinite;
&#125;

	</code>
